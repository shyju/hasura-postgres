This a sample node app which exposes an API which can be called through Hasura actions.

Try out the below mutation to trigger the insertProduct action from GraphiQL.


            // insert product
            mutation {
              insertProduct(productName: "ProductName") {
                affected_rows
              }
            }
          


            // send SMS 
            mutation {
              sendSMS(
                body: "// type your message here", 
                from: "+16066444698", // use this number as twilio is in trial mode
                to: "+919539394104" // can send to only one number in trial mode
              ) {
                response
              }
            }
